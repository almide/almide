// Almide IoError — structured filesystem error type
#[derive(Debug, Clone, PartialEq)]
struct AlmideIoError {
    message: String,
}
impl AlmideIoError {
    fn new(msg: impl Into<String>) -> Self { AlmideIoError { message: msg.into() } }
}
impl From<std::io::Error> for AlmideIoError {
    fn from(e: std::io::Error) -> Self { AlmideIoError { message: e.to_string() } }
}
impl std::fmt::Display for AlmideIoError {
    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result { write!(f, "{}", self.message) }
}
// Allow ? to convert AlmideIoError → String in effect fn returning Result<T, String>
impl From<AlmideIoError> for String {
    fn from(e: AlmideIoError) -> Self { e.message }
}
